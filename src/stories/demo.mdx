# Nanoui

# Install
```javascript
npm i nanoui
```

# Demo
```javascript
import React from 'react';
import { Button, SimpleButton } from 'nanoui';

export default function App() {
  let color1 = '#730707';
  return (
    <div>
      <Button
        backgroundColor={color1}
        label="Button"
        onClick={function noRefCheck() {}}
        primary
      />
      <SimpleButton>Hello</SimpleButton>
    </div>
  );
}

```
